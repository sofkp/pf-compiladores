struct Libro {
    titulo:&str,
    autores:[&str;3]
}

fn main() {
    let l = Libro {
        titulo:"Compilers 101",
        autores:["Aho","Sethi","Ullman"]
    };

    println!("titulo");
    println!("{}", l.titulo);
    println!("autor 1");
    println!("{}", l.autores[1]);

    l.autores[1] = "Julio Yarasca";
    println!("autor 1 nuevo");
    println!("{}", l.autores[1]);
}
